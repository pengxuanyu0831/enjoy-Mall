public class cn.enjoy.sys.model.TreeViewResource implements Serializable{

    private String id; //
    private String pid; //
    private String text; //
    private [] nodes; //

    public String getId(){
        return id;
    }

    public void setId(String id){
        this.id = id;
    }

    public String getPid(){
        return pid;
    }

    public void setPid(String pid){
        this.pid = pid;
    }

    public String getText(){
        return text;
    }

    public void setText(String text){
        this.text = text;
    }

    public [] getNodes(){
        return nodes;
    }

    public void setNodes([] nodes){
        this.nodes = nodes;
    }

}
